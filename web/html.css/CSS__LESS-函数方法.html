                # 颜色定义 (Color Definition)

                // 色相(H)、饱和度(S)、明度(L)
hsl(90, 100%, 50%) // => #80ff00 

// 色相(H)、饱和度(S)、明度(L)、透明度(A)
hsla(90, 100%, 50%, 0.5) // => rgba(128, 255, 0, 0.5)

// 色调（H），饱和度（S），明度（V）
hsv(90, 100%, 50%) // => #408000

// 提取色相
hue(hsl(90, 100%, 50%)) // => 90

                # 颜色操作

                // 增加饱和度
saturate(color, 20%)

// 减少饱和度
desaturate(color, 20%) 

// 增加亮度
lighten(color, 20%) 

// 减少亮度
darken(color, 20%) 

// 降低透明度
fadein(color, 10%) 

// 增加透明度
fadeout(color, 10%) 

// 设置绝对不透明度(覆盖原透明度)
fade(color, 20%) 

// 旋转色调角度
spin(color, 10) 

// 将两种颜色混合，不透明度包括在计算中。
mix(#f00, #00f, 50%) 

// 与白色混合
tint(#007fff, 50%) 

// 与黑色混合
shade(#007fff, 50%) 

// 灰度，移除饱和度
greyscale(color) 

// 返回对比度最大的颜色
contrast(color1, color2) 

// 颜色混合 每个RGB 通道相乘，然后除以255
multiply(color1, color2); 

// 与 multiply 相反
screen(color1, color2); 

// 使之更浅或更暗
overlay(color1, color2) 

// 避免太亮或太暗
softlight(color1, color2) 

// 与overlay相同，但颜色互换
hardlight(color1, color2) 

// 计算每个通道(RGB)基础上的两种颜色的平均值
average(color1, color2)
